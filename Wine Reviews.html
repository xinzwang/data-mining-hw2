def collect_has_row(data,pattern):
	"""data  pattern"""
	a = data.copy()
	pattern = [pattern] if isinstance(pattern, str) else pattern
	for p in pattern:
		a = a[a[col_name(p)[0]].str.contains(p)]	# 
	return a

def collect_not_row(data,pattern):
	"""data  pattern"""
	a = data.copy()
	pattern = [pattern] if isinstance(pattern, str) else pattern
	for p in pattern:
		a = a[~a[col_name(p)[0]].str.contains(p)]	# 
	return a

def lift(data):
	"""lift"""
	n = np.sum(data)
	a,b,c,d = data.reshape(-1)
	return (a * n)/((a+c)*(a+b))

def ch2(data):
	""""""
	n = np.sum(data)
	a ,b,c,d = data.reshape(-1)
	return n*np.square(a*d-b*c) / ((a+b)*(c+d)*(a+c)*(b+d))
